👉 Which programming lang has large user base ?¶

In [94]:
lang = data['LanguageWorkedWith']
# print(lang[: 5])

# as we can see we have lang seprated by a delimeter, we need a way to count each lang.

from collections import Counter

# to store the lang_count
lang_counter = Counter()

# we also have some nan entries i.e float here, so we have to check the type before spliting by a delimeter

for l in lang:
    if type(l) == str:
        lang_counter.update(l.split(';'))
    else:
        pass
        
# Let's plot what we have got

plt.rcParams['figure.figsize'] = [8, 6]
X = []
Y = []

for i in lang_counter.most_common():
    X.append(i[0])
    Y.append(i[1])

# reverse : to see most popular  on top
X.reverse()
Y.reverse()
fig,ax = plt.subplots()
plt.style.use('ggplot')
ax.set_title("Most Popular Prog Lang")
ax.barh(X,Y)
plt.tight_layout()
plt.show()
